import { useState, useEffect } from 'react';
import { DecayStatistics } from '@/types/memory';

export const useDecayStatistics = (userId: string) => {
  const [statistics, setStatistics] = useState<DecayStatistics | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetchStatistics();
  }, [userId]);

  const fetchStatistics = async () => {
    try {
      setLoading(true);
      const response = await fetch(`/api/v1/decay/statistics?user_id=${userId}`);
      const data = await response.json();
      
      if (data.success) {
        setStatistics(data.statistics);
        setError(null);
      } else {
        setError('Failed to fetch statistics');
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Unknown error');
    } finally {
      setLoading(false);
    }
  };

  return { statistics, loading, error, refetch: fetchStatistics };
};

export const useDecayedMemories = (userId: string, threshold: number = 0.3) => {
  const [memories, setMemories] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchDecayedMemories();
  }, [userId, threshold]);

  const fetchDecayedMemories = async () => {
    try {
      setLoading(true);
      const response = await fetch(
        `/api/v1/decay/decayed-memories?user_id=${userId}&threshold=${threshold}&limit=50`
      );
      const data = await response.json();
      
      if (data.success) {
        setMemories(data.memories);
      }
    } catch (error) {
      console.error('Failed to fetch decayed memories:', error);
    } finally {
      setLoading(false);
    }
  };

  return { memories, loading, refetch: fetchDecayedMemories };
};
